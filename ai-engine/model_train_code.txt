# STEP - 02
from unsloth import FastLanguageModel
import torch

model, tokenizer = FastLanguageModel.from_pretrained(
    model_name = "unsloth/meta-llama-3.1-8b-bnb-4bit", # The "Gold Standard" model
    max_seq_length = 2048,
    load_in_4bit = True,
)

model = FastLanguageModel.get_peft_model(
    model,
    r = 32, 
    target_modules = ["q_proj", "k_proj", "v_proj", "o_proj",
                      "gate_proj", "up_proj", "down_proj",],
    lora_alpha = 32,
    lora_dropout = 0,
    bias = "none",
)




# STEP - 03
from datasets import load_dataset

# Define the prompt style
alpaca_prompt = """Below is an instruction that describes a task. Write a response that appropriately completes the request.

### Instruction:
{}

### Response:
{}"""

def formatting_prompts_func(examples):
    instructions = examples["instruction"]
    inputs       = examples["input"]
    outputs      = examples["output"]
    texts = []
    for instruction, input, output in zip(instructions, inputs, outputs):
        # Combine instruction and input for a unified question
        full_instruction = f"{instruction} ({input})".strip() if input else instruction
        text = alpaca_prompt.format(full_instruction, output) + tokenizer.eos_token
        texts.append(text)
    return { "text" : texts, }

dataset = load_dataset("json", data_files="university_dataset.json", split="train")
dataset = dataset.map(formatting_prompts_func, batched = True)





# STEP - 04
from trl import SFTTrainer
from transformers import TrainingArguments

trainer = SFTTrainer(
    model = model,
    tokenizer = tokenizer,
    train_dataset = dataset,
    dataset_text_field = "text",
    max_seq_length = 2048,
    args = TrainingArguments(
        per_device_train_batch_size = 2,
        gradient_accumulation_steps = 4,
        warmup_steps = 5,
        max_steps = 150, # Llama-3.1 learns fast; 100-150 steps is usually enough
        learning_rate = 2e-4,
        fp16 = not torch.cuda.is_bf16_supported(),
        bf16 = torch.cuda.is_bf16_supported(),
        logging_steps = 1,
        optim = "adamw_8bit",
        weight_decay = 0.01,
        output_dir = "outputs",
    ),
)

trainer.train()

import litserve as ls
from unsloth import FastLanguageModel
import torch

class AdamasAIAPI(ls.LitAPI):
    def setup(self, device):
        # 1. Load the model using the files you uploaded
        # Make sure the 'model_name' matches the folder where your files are!
        self.model, self.tokenizer = FastLanguageModel.from_pretrained(
            model_name = "fine_tuned_model", 
            max_seq_length = 2048,
            load_in_4bit = True,
        )
        FastLanguageModel.for_inference(self.model)

    def decode_request(self, request):
        # 2. Extract the question from the JSON body
        return request.get("question", "")

    def predict(self, question):
        # 3. Format the prompt for Llama-3.1
        prompt = f"### Instruction:\n{question}\n\n### Response:\n"
        inputs = self.tokenizer([prompt], return_tensors = "pt").to("cuda")
        
        # Generate the answer
        outputs = self.model.generate(**inputs, max_new_tokens = 512)
        return self.tokenizer.batch_decode(outputs, skip_special_tokens=True)[0]

    def encode_response(self, output):
        # 4. Clean the text to send back only the Answer
        try:
            answer = output.split("### Response:")[1].strip()
        except:
            answer = output
        return {"answer": answer}

if __name__ == "__main__":
    # 5. Start the server on Port 8000
    api = AdamasAIAPI()
    server = ls.LitServer(api, accelerator="gpu")
    server.run(port=8000)
